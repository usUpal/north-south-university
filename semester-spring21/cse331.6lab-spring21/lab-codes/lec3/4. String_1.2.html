.MODEL SMALL 
.STACK 100H 
.DATA 
    MY_NAME DB 'U','S',20h,"UPAL", '$' ; "UPAL" will hold the index of individual like 2,3,4,5 
                                   ; $ is the terminating char use to represent string unlike array 
     
    MSG DB 5 DUP('U', 'S'), '$' ; USUSUSUSUS                                
.CODE 
    MOV AX, @DATA                  ; load data segment 
    MOV DS, AX                     ; init data segment aka load data segment to the code segment 
     
     
     
    ;SI, DI, BP, BX =====> INDEX REGISTERS 
     
    MOV BX, 3  ; DECIMAL 3 
    MOV DL, MY_NAME[BX]      ; WORKING WITH INDEX REGISTERS 
    MOV DL, MY_NAME[BX+1] 
    MOV DL, MY_NAME[BX*3]  
    INT 21H 
     
    MOV AH,9          ;function #9 is for char string output 
    LEA DX, MY_NAME   ; output will load to the DX register || LEA = load effective address 
    INT 21H 
 
 
 
 
 
EXIT: 
MOV AH, 4CH 
INT 21H
