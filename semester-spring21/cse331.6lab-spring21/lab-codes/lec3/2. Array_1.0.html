.MODEL SMALL 
.STACK 100H 
.DATA  
 
       
    ELON_MASK DB 9 DUP (?)  ; 9 DUPLICATE SPACE FOR BLANK 
     
     
.CODE       
    MOV AX,@DATA     ; load data segment to the AX register 
    MOV DS,AX        ; now move ax to the DS register because we can't copy anything directly in to DS aka Data Segment Register 
     
     
    MOV ELON_MASK[0], 'E'  
    MOV ELON_MASK[1], 'L'  
    MOV ELON_MASK[2], 'O'  
    MOV ELON_MASK[3], 'N'  
    MOV ELON_MASK[4], 20h     ; SPACE  
    MOV ELON_MASK[5], 'M'  
    MOV ELON_MASK[6], 'A'  
    MOV ELON_MASK[7], 'S'  
    MOV ELON_MASK[8], 'K' 
     
    MOV AH, 2 
    MOV DL, ELON_MASK[0]  
    INT 21H 
    MOV DL, ELON_MASK[1]  
    INT 21H 
    MOV DL, ELON_MASK[2]  
    INT 21H 
    MOV DL, ELON_MASK[3]  
    INT 21H 
    MOV DL, ELON_MASK[4]  
    INT 21H 
    MOV DL, ELON_MASK[5]  
    INT 21H 
    MOV DL, ELON_MASK[6]  
    INT 21H 
    MOV DL, ELON_MASK[7]  
    INT 21H 
    MOV DL, ELON_MASK[8]  
    INT 21H 
     
     
     
     
 
EXIT:                ; Exit lebel 
MOV AH, 4CH 
INT 21H
