.MODEL SMALL  
.STACK 100H  
.DATA   
  
        
    MY_NAME DB ?, ?, ?, ?                ; 4 blank spaces 
    ARRAY DB 23H, 33, 5 DUP (?), 7H, 'g' ; u can put vaulues into array in any pattern  
      
      
.CODE        
    MOV AX,@DATA     ; load data segment to the AX register  
    MOV DS,AX        ; now move ax to the DS register because we can't copy anything directly in to DS aka Data Segment Register  
      
      
      
    MOV MY_NAME[0], 'E'    ; replace the define value with "E"  
    MOV MY_NAME[1], 'L'    ;            same  
    MOV MY_NAME[2], '0'    ;            same  
    MOV MY_NAME[3], 'N'    ;            same  
   
       
      
    MOV AH, 2   
    MOV DL, MY_NAME [0]    ; single key output  
    INT 21H  
    MOV DL, MY_NAME [1]    
    INT 21H  
    MOV DL, MY_NAME [2]    
    INT 21H    
    MOV DL, MY_NAME [3]    
    INT 21H  
      
      
      
  
EXIT:                ; Exit lebel  
MOV AH, 4CH  
INT 21H
