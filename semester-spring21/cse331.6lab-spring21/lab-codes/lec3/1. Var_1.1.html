.MODEL SMALL 
.STACK 100H 
.DATA 
    NUMBER1 DB 45H   ; DB = define byte[8bit] | DW = define word[16bit] 
    NUMBER2 DB 35H   ; 35h is the value which is declared to the number2 
    NUMBER3 DB ?     ; declaring blank variable 
    NUMBER4 DB ? 
.CODE 
    MOV AX,@DATA     ; load data segment to the AX register 
    MOV DS,AX        ; now move ax to the DS register because we can't copy anything directly in to DS aka Data Segment Register 
     
    ADD NUMBER1,1  ; ADD 1 TO THE NUMBER1 AND STORE IT TO THE NUMBER1 
    SUB NUMBER2,1  ; SUB 2 TO THE NUMBER1 AND STORE IT TO THE NUMBER2 
    ADD NUMBER3,10 ; NUMBER3 IS BLANK AND STORE THE VALUE OF THE SOURCE TO THE NUMBER3 VARIABLE  
     
    MOV DL, NUMBER1 ; MOVE NUMBER1'S DATA TO THE DL FOR O/P 
    MOV AH,2        ; OUTPUT FUNCTION 
    INT 21H         ; FUNCTION INTERRUPT   
     
    MOV DL, 0DH     ;CRET 
    INT 21H 
    MOV DL, 0DH     ;CRET 
    INT 21H 
     
    MOV AH, 1       ;INPUT 
    INT 21H          
    MOV NUMBER4,AL  ;user input stored in AL and we move AL's value to the blank var name NUMBER4 
     
     
 
EXIT:                ; Exit level 
MOV AH, 4CH 
INT 21H
